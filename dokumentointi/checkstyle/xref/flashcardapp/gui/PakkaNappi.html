3   import java.awt.Cursor;
4   import java.awt.Font;
5   import java.awt.event.MouseListener;
6   import java.awt.font.TextAttribute;
7   import java.util.HashMap;
8   import java.util.Map;
9   import javax.swing.JButton;
10  
11  /**
12   * Class for deck buttons.
13   * @see PakkaPaneeli
14   */
15  public class PakkaNappi extends JButton {
16  
17      public PakkaNappi(String pakanNimi, MouseListener mouseListener) {
18          super(pakanNimi);
19          super.setBorder(null);
20          super.setBorderPainted(false);
21          super.setContentAreaFilled(false);
22          super.setCursor(new Cursor(Cursor.HAND_CURSOR));
23          super.setAlignmentX(JButton.CENTER_ALIGNMENT);
24          super.addMouseListener(mouseListener);
25      }
26  
27      /**
28       * Method toggles underline on and off for the button.
29       */
30      public void vaihdaAlleviivausNapille() {
31          Font font = super.getFont();
32          Map<TextAttribute, Object> attribuutit = new HashMap(font.getAttributes());
33  
34          if (attribuutit.get(TextAttribute.UNDERLINE) instanceof Integer) {
35              attribuutit.put(TextAttribute.UNDERLINE, TextAttribute.UNDERLINE);
36          } else {
37              attribuutit.put(TextAttribute.UNDERLINE, TextAttribute.UNDERLINE_ON);
38          }
39  
40          super.setFont(font.deriveFont(attribuutit));
41      }
42      
43      public String getNimi() {
44          return super.getText();
45      }
46      
47      public void setNimi(String nimi) {
48          super.setText(nimi);
49      }
50  }
