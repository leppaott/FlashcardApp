16  /**
17   * Default view with a list of decks.
18   */
19  public class PakkaPaneeli extends JPanel {
20  
21      private final Kayttoliittyma gui;
22      private final FlashcardApp app;
23      private final EventHandler handler;
24      private final PakkaPopupMenu popup;
25  
26      public PakkaPaneeli(Kayttoliittyma gui, FlashcardApp app) {
27          super();
28          this.gui = gui;
29          this.app = app;
30          this.handler = new EventHandler();
31          this.popup = new PakkaPopupMenu(handler);
32  
33          super.setLayout(new BoxLayout(this, BoxLayout.X_AXIS));
34          super.add(Box.createHorizontalGlue());
35  
36          lisaaPaneeli();
37      }
38  
39      private JPanel lisaaPaneeli() {
40          JPanel paneeli = new JPanel();
41          paneeli.setLayout(new BoxLayout(paneeli, BoxLayout.Y_AXIS));
42          paneeli.setAlignmentX(JPanel.CENTER_ALIGNMENT);
43          paneeli.setAlignmentY(JPanel.TOP_ALIGNMENT);
44          paneeli.setMaximumSize(gui.getFrame().getSize());
45  
46          super.add(paneeli);
47          super.add(Box.createHorizontalGlue());
48          return paneeli;
49      }
50  
51      private int maxKomponenttejaSarakkeessa(int korkeus, int komponenttiKorkeus) {
52          if (korkeus == 0 || komponenttiKorkeus == 0) { //korkeus zero upon launch
53              return 10;
54          } //26 "Decks" height
55          return (korkeus - 26) / komponenttiKorkeus;
56      }
57  
58      /**
59       * Methods adds a new deck button for a given name.
60       * @param nimi deck name
61       */
62      public void lisaaPakkaNappi(String nimi) {
63          JPanel paneeli = (JPanel) super.getComponent(super.getComponentCount() - 2);
64          PakkaNappi nappi = new PakkaNappi(nimi, handler);
65  
66          int max = maxKomponenttejaSarakkeessa(gui.getFrame().getContentPane().getHeight(),
67                  nappi.getHeight() + 10);
68          if (paneeli.getComponentCount() >= max) {
69              paneeli = lisaaPaneeli();
70          }
71  
72          paneeli.add(Box.createRigidArea(new Dimension(0, 5)));
73          paneeli.add(nappi);
74          paneeli.add(Box.createRigidArea(new Dimension(0, 5)));
75      }
76  
77      private void tyhjennaPakoista() {
78          super.removeAll();
79          super.add(Box.createHorizontalGlue());
80          lisaaPaneeli();
81      }
82  
83      /**
84       * Method clears and fills the view
85       */
86      public void tayta() {
87          tyhjennaPakoista();
88          for (Pakka pakka : app.getPakat()) {
89              lisaaPakkaNappi(pakka.getNimi());
90          }
91          app.tallennaPakat();
92      }
93  
94      private class EventHandler implements ActionListener, MouseListener {
95  
96          @Override
97          public void actionPerformed(ActionEvent e) {
98              JMenuItem item = (JMenuItem) e.getSource();
99              if (item.getText().equals("Delete")) {
100                 int vastaus = JOptionPane.showConfirmDialog(gui.getFrame(),
101                         "Do you really want to delete this deck?", "Confirm",
102                         JOptionPane.YES_NO_OPTION, JOptionPane.PLAIN_MESSAGE);
103                 if (vastaus == JOptionPane.YES_OPTION) {
104                     app.poistaPakka(popup.getNappi().getNimi());
105                     gui.taytaPakkaPaneeli();
106                     gui.paivitaFrame();
107                 }
108             }
109         }
110 
111         @Override
112         public void mouseClicked(MouseEvent e) {
113             PakkaNappi nappi = (PakkaNappi) e.getSource();
114             if (e.getButton() == MouseEvent.BUTTON1) {
115                 if (app.haePakka(nappi.getNimi()).getKoko() > 0) {
116                     nappi.vaihdaAlleviivausNapille();
117                     gui.harjoitaPakkaa(nappi.getNimi());
118                 } else {
119                     JOptionPane.showMessageDialog(gui.getFrame(),
120                             "The deck is empty. Add cards before practicing.");
121                 }
122             } else if (e.getButton() == MouseEvent.BUTTON3) {
123                 popup.setNappi(nappi);
124                 popup.show(e.getComponent(), e.getX(), e.getY());
125             }
126         }
127 
128         @Override
129         public void mouseEntered(MouseEvent e) {
130             PakkaNappi nappi = (PakkaNappi) e.getSource();
131             nappi.vaihdaAlleviivausNapille();
132         }
133 
134         @Override
135         public void mouseExited(MouseEvent e) {
136             PakkaNappi nappi = (PakkaNappi) e.getSource();
137             nappi.vaihdaAlleviivausNapille();
138         }
139 
140         @Override
141         public void mousePressed(MouseEvent e) {
142         }
143 
144         @Override
145         public void mouseReleased(MouseEvent e) {
146         }
147     }
